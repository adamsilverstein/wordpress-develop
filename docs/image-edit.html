			x = this.intval( elX.val() ), y = this.intval( elY.val() ),

	/**
	 * Rounds a number to a whole.
	 *
	 * @memberof imageEdit
	 * @since    2.9.0
	 *
	 * @param {number} num The number.
	 *
	 * @returns {number} The number rounded to a whole number.
	 */
	round : function(num) {
		var s;
		num = Math.round(num);

		if ( this.hold.sizer > 0.6 ) {
			return num;
		}

		s = num.toString().slice(-1);

		if ( '1' === s ) {
			return num - 1;
		} else if ( '9' === s ) {
			return num + 1;
		}

		return num;
	},
